class Solution {

    TreeNode curr;
    public TreeNode increasingBST(TreeNode root) {
        
        TreeNode first = new TreeNode(-1);
        
        
        curr = first;
        
		
        dfs(root);
        
        return first.right;
    }
    
    private void dfs(TreeNode node){
        if(node == null) return;
        
        
        dfs(node.left);
        
        
        curr.right = new TreeNode(node.val);
        
        
        curr = curr.right;
        
        dfs(node.right);
    }
}
